```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
        
    int n, m;
    
    cin >> n;
    cin >> m;
    
    pair<int, float> arr[m];
    
    for(int i=0; i<m; i++)
        cin >> arr[i].first;
    
    cin >> m;
    
    for(int i=0; i<m; i++)
        cin >> arr[i].second;
            
    if(n==1) {
        cout << arr[0].second << endl;
        return 0;
    }
    
    int low = 0;
    int high = m-1;
    
    while(low<=high) {
        int mid = low + (high-low)/2;
        
        if(arr[mid].first<=n) {
             low = ++mid;
        } else {
            high = --mid;
        }
    }
    
    float x1, x2, ans;
    int y1, y2;
    
    if(arr[high].first==n) {
        
        ans = arr[high].second;
        cout << ans << endl;
        return 0;
        
    }else if(high>=0 && high<m-1) {
        
        x1 = arr[high].second;
        x2 = arr[high+1].second;
        y1 = arr[high].first;
        y2 = arr[high+1].first;
        
    } else if(high<0) {
        
        x1 = arr[0].second;
        x2 = arr[1].second;
        y1 = arr[0].first;
        y2 = arr[1].first;
        
    } else {
        
        x1 = arr[m-1].second;
        x2 = arr[m-2].second;
        y1 = arr[m-1].first;
        y2 = arr[m-2].first;            
    }
    
    ans = arr[high].second + (n-y1)*(x2-x1)/(y2-y1);
    
    cout << round(ans*100)/100;
    
    return 0;
}
```
