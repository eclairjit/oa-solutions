#include <bits/stdc++.h>
using namespace std;

class disjointSet {
    
  vector<int> rank;
  vector<int> parent;
  
  public:
  
    disjointSet(int n) {
        this -> rank.resize(n+1, 1);
        this -> parent.resize(n+1);
        
        for(int i=0; i<=n; i++)
            this -> parent[i] = i;
    }
    
    int getParent(int u) {
        
        if(this -> parent[u] == u) return u;
        return this -> parent[u] = getParent(this -> parent[u]);
        
    }
    
    void join(int u, int v) {
        
        int paretOfU = getParent(u);
        int paretOfV = getParent(v);
        
        if(paretOfU == paretOfV) return;
        
        if(this -> rank[paretOfU] >= this -> rank[paretOfV]) {
            this -> parent[paretOfV] = paretOfU;
            this -> rank[paretOfU] += this -> rank[paretOfV];
        } else {
            this -> parent[paretOfU] = paretOfV;
            this -> rank[paretOfV] += this -> rank[paretOfU];
        }
        
    }
    
};

int getMin(int n, string *otps) {
    
    vector<int> hash(26, -1);
    disjointSet ds(n);
    
    for(int i=0; i<n; i++) {
        
        string s = otps[i];
        
        for(int j=0; j<s.size(); j++) {
            
            if(hash[s[j]-'a']==-1) {
                hash[s[j]-'a'] = i;
            } else {
                int u = hash[s[j]-'a'];
                int v = i;
                
                ds.join(u, v);                
            }
            
        }        
         
    }
    
    int otpCount = 0;
    
    for(int i=0; i<n; i++) {
        
        if(ds.getParent(i)==i)
            otpCount++;
        
    }
    
    return otpCount;
    
}

int main() {
    
    int n;
    cin >> n;
    
    string otps[n];
    for(int i=0; i<n; i++)
        cin >> otps[i];
    
    cout << getMin(n, otps);
    
    return 0;
}
